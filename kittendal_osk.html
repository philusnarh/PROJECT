                #print t,f

                    if start > stop : step *= -1
                    f = np.arange(start , stop,step )

    def to_ephem_date(self,secs):
        import time
        """Convert timestamp to :class:`ephem.Date` object."""
        int_secs = np.floor(secs)
        timetuple = list(time.gmtime(int_secs)[:6])
        timetuple[5] += secs - int_secs
        return ephem.Date(tuple(timetuple))

    def _remove_duplicates(self,sensor):
        x = np.atleast_1d(sensor['timestamp'])
        y = np.atleast_1d(sensor['value'])
        z = np.atleast_1d(sensor['status'])
        # Sort x via mergesort, as it is usually already sorted and stability is important
        sort_ind = np.argsort(x, kind='mergesort')
        x, y = x[sort_ind], y[sort_ind]
        # Array contains True where an x value is unique or the last of a run of identical x values
        last_of_run = np.asarray(list(np.diff(x) != 0) + [True])
        # Discard the False values, as they represent duplicates - simultaneously keep last of each run of duplicates
        unique_ind = last_of_run.nonzero()[0]
        # Determine the index of the x value chosen to represent each original x value (used to pick y values too)
        replacement = unique_ind[len(unique_ind) - np.cumsum(last_of_run[::-1])[::-1]]
        # All duplicates should have the same y and z values - complain otherwise, but continue
        return (x[unique_ind], y[unique_ind])

    def _get_sensor_arr(self,h5ref,timestamps,order=0,true_equals=None):
        timevals,valuevals =  self._remove_duplicates(h5ref)
        if true_equals is None :
            valuevals = valuevals.astype(float)
        else :
            valuevals =   valuevals==true_equals
        return fit.PiecewisePolynomial1DFit(order).fit(timevals,valuevals)(timestamps[:])


    def get_noise_diode(self,corr_products=None,sensordata=None,noise_type='coupler',dump_period=None): #
        if corr_products is None:corr_products =self.corrprods
        if sensordata is None:  sensordata=self.file
        if dump_period is None: dump_period = self.dump_period

        ants = set([x[:-1] for x in set(corr_products.flatten())])
        noise_diode = {}
        for ant in ants:
            timevals,valuevals =  self._remove_duplicates(sensordata['MetaData/Sensors/Antennas/%s/rfe3.rfe15.noise.%s.on' % (ant,noise_type) ])
            #print ant , 
            valuevals = valuevals.astype(float)
            time_lookup = fit.PiecewisePolynomial1DFit(0).fit(timevals-0.5*dump_period,timevals)
            noise_diodeval = fit.PiecewisePolynomial1DFit(0).fit(timevals,valuevals)
            def temp_func(timey):
                valfactor = (time_lookup(timey)-(timey-0.5*dump_period))/(dump_period) 
                # (noise_time - dump_start)/dump_period 
                valfactor = np.clip(valfactor,0.0,1.0)
                valfactor[(noise_diodeval(timey-0.5*dump_period) == 1) * (noise_diodeval(timey+0.5*dump_period) == 1)]  = 1 # 100% on
                valfactor[(noise_diodeval(timey-0.5*dump_period) == 0) * (noise_diodeval(timey+0.5*dump_period) == 1)]  *=  -1 # swithcing on
                valfactor[(noise_diodeval(timey-0.5*dump_period) == 0) * (noise_diodeval(timey+0.5*dump_period) == 1)]  +=  1  
                return valfactor
            noise_diode[ant] =  temp_func(self.timestamps)
        return noise_diode

    def plot_data(self,x,y,z,gsize=100,scatter=False):
        """Plotting function
        This plots a rasterscan as an intensity image
        the scatter parameter adds markers to indecate the data points
        x,y,z must be the same length and z is the amplitude"""
        from scipy.interpolate import griddata
        import matplotlib.pyplot as plt
        # define grid.
        npts = z.shape[0]
        xi = np.linspace(x.min(),x.max(),gsize)
        yi = np.linspace(y.min(),y.max(),gsize)
        # grid the data.
        zi = griddata((x, y), z, (xi[None,:], yi[:,None]))
        # contour the gridded data, plotting dots at the randomly spaced data points.
        CS = plt.contour(xi,yi,zi,15,linewidths=0.5,colors='k')
        CS = plt.contourf(xi,yi,zi,15,cmap=plt.cm.jet)
        plt.colorbar() # draw colorbar
        # plot data points.
        if scatter :
            plt.scatter(x,y,marker='o',c='b',s=5)
        plt.xlim(x.min(),x.max())
        plt.ylim(y.min(),y.max())
        plt.title('griddata test (%d points)' % npts)

def plot_data(self,x,y,z,gsize=100,scatter=False):
        """Plotting function
        This plots a rasterscan as an intensity image
        the scatter parameter adds markers to indecate the data points
        x,y,z must be the same length and z is the amplitude"""
        from scipy.interpolate import griddata
        import matplotlib.pyplot as plt
        # define grid.
        npts = z.shape[0]
        xi = np.linspace(x.min(),x.max(),gsize)
        yi = np.linspace(y.min(),y.max(),gsize)
        # grid the data.
        zi = griddata((x, y), z, (xi[None,:], yi[:,None]))
        
        # contour the gridded data, plotting dots at the randomly spaced data points.
        CS = plt.contour(xi,yi,zi,15,linewidths=0.5,colors='k')
        CS = plt.contourf(xi,yi,zi,15,cmap=plt.cm.jet)
        plt.colorbar() # draw colorbar
        # plot data points.
        if scatter :
            plt.scatter(x,y,marker='o',c='b',s=5)
        plt.xlim(x.min(),x.max())
        plt.ylim(y.min(),y.max())
        plt.title('griddata test (%d points)' % npts)
